                submitBtn.disabled = true;
                
                // محاكاة الإرسال
                setTimeout(function() {

                    submitBtn.classList.add('btn-success');

                    successMsg.className = 'alert alert-success mt-4';
                    successMsg.innerHTML = `

                    `;
                    form.parentNode.insertBefore(successMsg, form.nextSibling);
